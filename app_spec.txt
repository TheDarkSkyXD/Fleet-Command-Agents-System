<project_specification>
  <project_name>Fleet Command Agents System</project_name>

  <overview>
    Fleet Command Agents System is an Electron desktop application that brings multi-agent AI coding orchestration to a polished graphical interface. Based on Overstory's architecture, it replaces tmux-based terminal management with Electron-managed processes using node-pty, enabling users to spawn, coordinate, and monitor swarms of AI coding agents through a beautiful dark-themed UI. Powered by Claude Code CLI subscription authentication, it requires no API keys — agents run through the user's existing Claude Code subscription.
  </overview>

  <database_schema>
    <tables>
      <sessions>
        - id (TEXT PRIMARY KEY), run_id, agent_name, capability, state (booting|working|completed|stalled|zombie)
        - pid, worktree_path, branch_name, task_id, parent_agent, depth
        - tmux_session (legacy, unused in Electron), transcript_path, prompt_version
        - escalation_level, stalled_at, created_at, updated_at, completed_at
      </sessions>

      <runs>
        - id (TEXT PRIMARY KEY), status (active|completed|failed)
        - coordinator_session_id, agent_count
        - started_at, completed_at
      </runs>

      <messages>
        - id (TEXT PRIMARY KEY), thread_id, from_agent, to_agent
        - subject, body, type (status|question|result|error|worker_done|merge_ready|merged|merge_failed|escalation|health_check|dispatch|assign)
        - priority (low|normal|high|urgent), payload (JSON), read (INTEGER)
        - created_at
      </messages>

      <events>
        - id (TEXT PRIMARY KEY), run_id, agent_name, session_id
        - event_type (tool_start|tool_end|session_start|session_end|mail_sent|mail_received|spawn|error|custom)
        - tool_name, tool_args, tool_duration_ms, level (debug|info|warn|error)
        - data (JSON), created_at
      </events>

      <metrics>
        - id (TEXT PRIMARY KEY), agent_name, task_id, capability
        - input_tokens, output_tokens, cache_read_tokens, cache_creation_tokens
        - model_used, estimated_cost, duration_ms
        - parent_agent, run_id
        - started_at, completed_at
      </metrics>

      <merge_queue>
        - id (INTEGER PRIMARY KEY AUTOINCREMENT), branch_name, task_id, agent_name
